 20class LearningAutomaton:
 21    """
 22    Implementacja automatu uczącego (Learning Automaton) typu L_R-I
 23    (Linear Reward-Inaction) dla dwóch akcji: ACTIVE i SLEEP.
 24
 25    Automat uczący służy sensorowi do dynamicznego dostosowywania
 26    prawdopodobieństwa wyboru akcji na podstawie otrzymywanych nagród
 27    (lub braku kary, w przypadku Inaction).
 28    """
 29    def __init__(self, num_actions=2, learning_rate_reward_A=0.1):
 30        """
 31        Konstruktor klasy LearningAutomaton.
 32
 33        Inicjalizuje automat uczący z określoną liczbą akcji i parametrem uczenia 'a'.
 34
 35        Args:
 36            num_actions (int): Liczba dostępnych akcji (domyślnie 2: ACTIVE, SLEEP).
 37                               Ta implementacja wspiera tylko 2 akcje.
 38            learning_rate_reward_A (float): Parametr uczenia 'a' dla nagród
 39                                           (musi być w zakresie (0, 1)).
 40        """
 41        assert num_actions == 2, "LearningAutomaton tylko dla 2 akcji (ACTIVE, SLEEP)."
 42        self.num_actions = num_actions
 43        self.a_param = learning_rate_reward_A
 44        self.action_probabilities = [0.5, 0.5]
 45        self.chosen_action_index = None
 46
 47    def set_probabilities_based_on_energy_ratio(self, energy_ratio: float):
 48        """
 49        Ustawia prawdopodobieństwa wyboru akcji ACTIVE/SLEEP na podstawie
 50        aktualnego wskaźnika energii sensora.
 51
 52        Prawdopodobieństwo bycia ACTIVE jest ustawiane proporcjonalnie do
 53        wskaźnika energii (pozostała energia sensora / całkowita energia sieci),
 54        a prawdopodobieństwo bycia SLEEP jest dopełnieniem do 1.0.
 55
 56        Args:
 57            energy_ratio (float): Wskaźnik energii sensora, wartość w zakresie [0.0, 1.0].
 58                                  Obliczana w klasie Network (faza Learning).
 59        """
 60        prob_active = max(0.0, min(1.0, energy_ratio))
 61        self.action_probabilities[0] = prob_active
 62        self.action_probabilities[1] = 1.0 - prob_active
 63        self._normalize_and_clip()
 64    
 65    def choose_action(self) -> int:
 66        """
 67        Wybiera akcję (ACTIVE lub SLEEP) na podstawie aktualnych prawdopodobieństw.
 68
 69        Wykorzystuje ważony losowy wybór spośród dostępnych akcji.
 70
 71        Returns:
 72            int: Indeks wybranej akcji (0 dla ACTIVE, 1 dla SLEEP).
 73        """
 74        self._normalize_and_clip()
 75        self.chosen_action_index = random.choices(
 76            range(self.num_actions),
 77            weights=self.action_probabilities,
 78            k=1
 79        )[0]
 80        return self.chosen_action_index

 82    def update_probabilities_LRI(self, chosen_action_idx: int, is_reward_signal: bool):
 83        """
 84        Aktualizuje prawdopodobieństwa wyboru akcji zgodnie ze schematem L_R-I.
 85
 86        Aktualizacja następuje tylko w przypadku otrzymania sygnału nagrody (`is_reward_signal` == True).
 87        Zgodnie ze schematem L_R-I, tylko prawdopodobieństwo nagrodzonej akcji rośnie,
 88        a prawdopodobieństwa pozostałych akcji maleją (proporcjonalnie).
 89        W tej implementacji, nagroda (jeśli występuje) jest ZAWSZE za akcję ACTIVE.
 90
 91        Args:
 92            chosen_action_idx (int): Indeks akcji, która "została podjęta"
 93                                     i jest oceniana (dla algorytmu CS, to akcja ACTIVE sensora).
 94                                     Teoretycznie może być użyty do nagradzania innych akcji,
 95                                     ale w tym kontekście nagradzamy tylko bycie ACTIVE.
 96            is_reward_signal (bool): True, jeśli sensor otrzymał sygnał nagrody
 97                                     (np. za bycie częścią dobrego CS); False w przeciwnym razie.
 98        """
 99        if not is_reward_signal:
100            return
101
102        action_to_reward_idx = Sensor.ACTION_ACTIVE_IDX # Zawsze nagradzamy akcję ACTIVE
103        
104        if chosen_action_idx != action_to_reward_idx:
105            logging.warning(f"LA Update: Rewarding action {chosen_action_idx}, but expected to reward ACTION_ACTIVE_IDX {action_to_reward_idx}.")
106
107        other_action_idx = Sensor.ACTION_SLEEP_IDX
108
109        current_prob_rewarded = self.action_probabilities[action_to_reward_idx]
110        current_prob_other = self.action_probabilities[other_action_idx]
111
112        # Formuła aktualizacji prawdopodobieństw dla L_R-I (jeśli jest nagroda):
113        # Prawdopodobieństwo nagrodzonej akcji rośnie: p_i(t+1) = p_i(t) + a * (1 - p_i(t))
114        self.action_probabilities[action_to_reward_idx] = current_prob_rewarded + \
115                                                          self.a_param * (1.0 - current_prob_rewarded)
116        # Prawdopodobieństwa pozostałych akcji maleją proporcjonalnie: p_j(t+1) = p_j(t) - a * p_j(t)
117        self.action_probabilities[other_action_idx] = current_prob_other - \
118                                                      self.a_param * current_prob_other
119        
120        self._normalize_and_clip()
121
122    def _normalize_and_clip(self):
123        """
124        Normalizuje sumę prawdopodobieństw do 1.0 i przycina wartości,
125        aby mieściły się w zakresie [min_prob, 1 - min_prob].
126
127        Zapobiega to zbieganiu prawdopodobieństw do 0 lub 1 zbyt szybko.
128        Minimalne prawdopodobieństwo zapewnia, że każda akcja ma zawsze
129        pewną (choćby małą) szansę na wybór.
130        """
131        min_prob = 0.001
132        p0 = self.action_probabilities[Sensor.ACTION_ACTIVE_IDX]
133        p1 = self.action_probabilities[Sensor.ACTION_SLEEP_IDX]
134
135        # Krok 1: Upewnij się, że indywidualne prawdopodobieństwa są w zakresie [0, 1]
136        p0 = max(0.0, min(1.0, p0))
137        p1 = max(0.0, min(1.0, p1))
138        
139        # Krok 2: Normalizacja sumy do 1.0
140        current_total_prob = p0 + p1
141        if abs(current_total_prob - 1.0) > 1e-9: # Jeśli suma znacząco odbiega od 1
142            if current_total_prob == 0:
143                self.action_probabilities = [0.5] * self.num_actions
144            else:
145                self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = p0 / current_total_prob
146                self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = p1 / current_total_prob
147        else:
148            self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = p0
149            self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = p1
150
151        # Krok 3: Zastosuj min_prob, zachowując sumę równą 1
152        if self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] < min_prob:
153            self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = min_prob
154            self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = 1.0 - min_prob
155        elif self.action_probabilities[Sensor.ACTION_SLEEP_IDX] < min_prob: # elif jest kluczowe
156            self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = min_prob
157            self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = 1.0 - min_prob
158        
159        # Ostateczne sprawdzenie i ewentualna normalizacja (głównie na wypadek, gdyby min_prob > 0.5)
160        final_sum = sum(self.action_probabilities)
161        if abs(final_sum - 1.0) > 1e-9:
162            if final_sum == 0: self.action_probabilities = [0.5] * self.num_actions
163            else: self.action_probabilities = [p / final_sum for p in self.action_probabilities]
164
165    def initialize_probabilities(self):
166        """
167        Resetuje prawdopodobieństwa wyboru akcji do wartości początkowych (np. 0.5 dla każdej akcji).
168
169        Wywoływana w fazie Network Setup na początku symulacji.
170        """
171        self.action_probabilities = [0.5, 0.5]  # Reset probabilities to default values
172        self.chosen_action_index = None  # Reset chosen action index

179class Sensor:
180    """
181    Reprezentuje pojedynczy węzeł sensorowy w sieci WSN.
182
183    Klasa Sensor przechowuje informacje o identyfikatorze sensora,
184    jego pozycji, poziomie energii, zasięgach komunikacji i wykrywania.
185    Zarządza stanem operacyjnym sensora (ACTIVE, SLEEP, DEAD), buforem
186    danych oraz interakcją z przypisanym automatem uczącym (Learning Automaton)
187    do podejmowania decyzji o stanie aktywności w każdej rundzie.
188    """
189    # Definicje indeksów akcji dla LA
190    ACTION_ACTIVE_IDX = 0
191    ACTION_SLEEP_IDX = 1
192
193    def __init__(self, id, x, y, initial_energy, comm_range, sensing_range, sink_id=None,
194                 learning_rate_reward_A=0.1):
195        """
196        Konstruktor klasy Sensor.
197
198        Inicjalizuje sensor z podstawowymi parametrami fizycznymi i energetycznymi.
199        Przypisywana jest mu rola stacji bazowej (sink) lub zwykłego sensora.
200        Zwykłym sensorom przypisywany jest automat uczący.
201
202        Args:
203            id (int): Unikalny identyfikator sensora.
204            x (float): Współrzędna X pozycji sensora.
205            y (float): Współrzędna Y pozycji sensora.
206            initial_energy (float): Początkowy poziom energii sensora.
207            comm_range (float): Zasięg komunikacji sensora.
208            sensing_range (float): Zasięg sensoryczny (wykrywania) sensora.
209            sink_id (int | None): ID stacji bazowej. Jeśli ID sensora jest równe sink_id,
210                                  sensor jest traktowany jako stacja bazowa (domyślnie None).
211            learning_rate_reward_A (float): Parametr uczenia 'a' dla automatu LA tego sensora.
212                                           Używane tylko dla sensorów niebędących sinkiem.
213        """
214        self.id = id
215        self.pos = (x, y)
216        self.initial_energy = initial_energy
217        self.current_energy = initial_energy
218        self.comm_range = comm_range
219        self.sensing_range = sensing_range
220        self.state = SensorState.SLEEP
221        self.is_sink = (self.id == sink_id)
222        self.is_critical_sensor = False
223        self.neighbor_poi_coverage: dict[int, set[int]] = {}
224        self.time_last_heard_from_neighbor: dict[int, int] = {}
225
226        if self.is_sink:
227            self.state = SensorState.ACTIVE
228            self.current_energy = float('inf')
229
230        self.la = LearningAutomaton(
231            learning_rate_reward_A=learning_rate_reward_A
232        ) if not self.is_sink else None
233
234        if not self.is_sink and self.la is None:
235            raise ValueError(f"Sensor {self.id} must have a Learning Automaton assigned.")
236        
237        self.last_la_action_idx = None
238
239        self.neighbors = []
240        self.monitored_pois = []
241        self.parent_to_sink = None
242        self.data_buffer = []
243        self.is_failed = False
244        self.is_critical_sensor = False

246    def distance_to(self, target_pos_or_sensor):
247        """
248        Oblicza odległość Euclideanową do innego obiektu (sensora lub POI)
249        lub do podanej pozycji (x, y).
250
251        Args:
252            target_pos_or_sensor (Union[tuple[float, float], 'Sensor']): Pozycja docelowa
253                                                                         (x, y) lub obiekt
254                                                                         posiadający atrybut `.pos`.
255
256        Returns:
257            float: Odległość między bieżącym sensorem a celem.
258        """
259        if hasattr(target_pos_or_sensor, 'pos'):
260            target_pos = target_pos_or_sensor.pos
261        else:
262            target_pos = target_pos_or_sensor
263        return math.sqrt((self.pos[0] - target_pos[0])**2 + (self.pos[1] - target_pos[1])**2)

265    def can_communicate_with(self, other_sensor):
266        """
267        Sprawdza, czy bieżący sensor może potencjalnie komunikować się
268        z innym sensorem.
269
270        Potencjalna komunikacja jest możliwa, jeśli oba sensory nie są trwale
271        uszkodzone i znajdują się w zasięgu komunikacji siebie nawzajem.
272        Ta metoda nie sprawdza, czy sensor `other_sensor` jest w stanie ACTIVE
273        (co jest wymagane do *rzeczywistego* odebrania pakietu), a jedynie
274        fizyczną/topologiczną możliwość połączenia.
275
276        Args:
277            other_sensor ('Sensor'): Obiekt innego sensora do sprawdzenia.
278
279        Returns:
280            bool: True, jeśli potencjalna komunikacja jest możliwa, False w przeciwnym przypadku.
281        """
282        if other_sensor.is_failed or self.is_failed:
283            return False
284        return self.distance_to(other_sensor.pos) <= self.comm_range and self.id != other_sensor.id
285
286    def can_sense_poi(self, poi):
287        """
288        Sprawdza, czy bieżący sensor może sensorycznie wykryć dany punkt POI.
289
290        Wykrywanie jest możliwe, jeśli sensor nie jest trwale uszkodzony,
291        nie jest martwy i POI znajduje się w jego zasięgu sensorycznym.
292        Nie sprawdza, czy sensor jest w stanie ACTIVE (ta metoda określa
293        *potencjał* wykrywania).
294
295        Args:
296            poi ('POI'): Obiekt punktu zainteresowania (POI) do sprawdzenia.
297
298        Returns:
299            bool: True, jeśli POI jest w zasięgu sensorycznym sensora, False w przeciwnym przypadku.
300        """
301        if self.is_failed or self.state == SensorState.DEAD:
302            return False
303        return self.distance_to(poi.pos) <= self.sensing_range

305    def update_energy(self, activity_type=None, amount=None, duration=1.0):
306        """
307        Aktualizuje poziom energii sensora na podstawie zużycia.
308
309        Energia może być zredukowana o stałą `amount` lub na podstawie
310        zdefiniowanych kosztów energetycznych dla określonego typu aktywności
311        i czasu trwania. Jeśli energia spadnie do zera lub poniżej, stan sensora
312        zmienia się na DEAD.
313
314        Args:
315            activity_type (SensorState | str | None): Typ aktywności, która powoduje zużycie energii
316                                                    (np. SensorState.ACTIVE, SensorState.SLEEP, "TRANSMIT", "RECEIVE", "PROCESSING").
317                                                    Jeśli None i `amount` też None, zużycie zależy od aktualnego stanu.
318            amount (float | None): Stała ilość energii do odjęcia. Jeśli podane,
319                                   przesłania zużycie oparte na activity_type.
320            duration (float): Czas trwania aktywności (domyślnie 1.0, np. reprezentuje 1 rundę).
321        """
322        if self.is_sink or self.state == SensorState.DEAD: return
323        cost = 0
324        if amount: cost = amount
325        elif activity_type:
326            if activity_type == SensorState.ACTIVE: cost = EnergyConsumption.MONITORING * duration
327            elif activity_type == SensorState.SLEEP: cost = EnergyConsumption.SLEEP * duration
328            elif activity_type == "PROCESSING": cost = EnergyConsumption.PROCESSING * duration
329        else:
330            if self.state == SensorState.ACTIVE: cost = EnergyConsumption.MONITORING * duration
331            elif self.state == SensorState.SLEEP: cost = EnergyConsumption.SLEEP * duration
332        self.current_energy -= cost
333        if self.current_energy <= 0:
334            self.current_energy = 0
335            self.state = SensorState.DEAD

337    def handle_broadcast_message(self, sender_id: int, message_type: str, payload: any, network_time: int):
338        """
339        Przetwarza otrzymaną wiadomość broadcastową od innego sensora.
340
341        Sensory używają tego do aktualizacji swoich lokalnych informacji, np.
342        o pokryciu POI przez sąsiadów (na podstawie komunikatów "POI_COVERAGE_ADVERTISEMENT").
343
344        Args:
345            sender_id (int): ID sensora, który wysłał wiadomość.
346            message_type (str): Typ otrzymanej wiadomości (np. "POI_COVERAGE_ADVERTISEMENT").
347            payload (any): Treść (ładunek) wiadomości.
348            network_time (int): Aktualna runda symulacji (czas sieciowy),
349                                w której wiadomość została odebrana.
350        """
351        if message_type == "POI_COVERAGE_ADVERTISEMENT":
352            if isinstance(payload, dict) and 'covered_poi_ids' in payload:
353                covered_ids = set(payload['covered_poi_ids'])
354                self.neighbor_poi_coverage[sender_id] = covered_ids
355                self.time_last_heard_from_neighbor[sender_id] = network_time
356                # logging.debug(f"  Sensor {self.id} updated POI coverage from neighbor {sender_id}: {covered_ids}")
357            else:
358                logging.warning(f"Sensor {self.id} received malformed POI_COVERAGE_ADVERTISEMENT from {sender_id}")
359        
360        elif message_type == "NEIGHBOR_ANNOUNCEMENT":
361            pass

363    def __repr__(self):
364        """
365        Zwraca reprezentację sensora jako string (do celów debugowania/logowania).
366
367        Pokazuje ID sensora, poziom energii, aktualny stan, informację
368        czy jest sensorem krytycznym, prawdopodobieństwo P(ACTIVE) LA (jeśli posiada)
369        oraz opcjonalnie ID pokrywanych POI (jeśli jest aktywny i coś pokrywa).
370        """
371        la_probs_str = ""
372        if self.la:
373            la_probs_str = f"LA_P(A):{self.la.action_probabilities[Sensor.ACTION_ACTIVE_IDX]:.2f}"
374        crit_str = " CRIT" if self.is_critical_sensor else ""
375        covered_pois_repr = ""
376        if self.state == SensorState.ACTIVE and self.monitored_pois:
377            monitored_ids = {p.id for p in self.monitored_pois}
378            if monitored_ids:
379                covered_pois_repr = f" COV:{monitored_ids}"
380                
381        return (f"Sensor(id={self.id}, E:{self.current_energy:.2f}, S:{self.state}{crit_str}{covered_pois_repr}, {la_probs_str})"
382                if not self.is_sink else
383                f"Sensor(id={self.id}, S:{self.state}) (SINK)")

 83    def update_probabilities_LRI(self, chosen_action_idx: int, is_reward_signal: bool):
 84        """
 85        Aktualizuje prawdopodobieństwa wyboru akcji zgodnie ze schematem L_R-I.
 86
 87        Aktualizacja następuje tylko w przypadku otrzymania sygnału nagrody (`is_reward_signal` == True).
 88        Zgodnie ze schematem L_R-I, tylko prawdopodobieństwo nagrodzonej akcji rośnie,
 89        a prawdopodobieństwa pozostałych akcji maleją (proporcjonalnie).
 90        W tej implementacji, nagroda (jeśli występuje) jest ZAWSZE za akcję ACTIVE.
 91
 92        Args:
 93            chosen_action_idx (int): Indeks akcji, która "została podjęta"
 94                                     i jest oceniana (dla algorytmu CS, to akcja ACTIVE sensora).
 95                                     Teoretycznie może być użyty do nagradzania innych akcji,
 96                                     ale w tym kontekście nagradzamy tylko bycie ACTIVE.
 97            is_reward_signal (bool): True, jeśli sensor otrzymał sygnał nagrody
 98                                     (np. za bycie częścią dobrego CS); False w przeciwnym razie.
 99        """
100        if not is_reward_signal:
101            return
102
103        action_to_reward_idx = Sensor.ACTION_ACTIVE_IDX # Zawsze nagradzamy akcję ACTIVE
104        
105        if chosen_action_idx != action_to_reward_idx:
106            logging.warning(f"LA Update: Rewarding action {chosen_action_idx}, but expected to reward ACTION_ACTIVE_IDX {action_to_reward_idx}.")
107
108        other_action_idx = Sensor.ACTION_SLEEP_IDX
109
110        current_prob_rewarded = self.action_probabilities[action_to_reward_idx]
111        current_prob_other = self.action_probabilities[other_action_idx]
112
113        # Formuła aktualizacji prawdopodobieństw dla L_R-I (jeśli jest nagroda):
114        # Prawdopodobieństwo nagrodzonej akcji rośnie: p_i(t+1) = p_i(t) + a * (1 - p_i(t))
115        self.action_probabilities[action_to_reward_idx] = current_prob_rewarded + \
116                                                          self.a_param * (1.0 - current_prob_rewarded)
117        # Prawdopodobieństwa pozostałych akcji maleją proporcjonalnie: p_j(t+1) = p_j(t) - a * p_j(t)
118        self.action_probabilities[other_action_idx] = current_prob_other - \
119                                                      self.a_param * current_prob_other
120        
121        self._normalize_and_clip()
122
123    def _normalize_and_clip(self):
124        """
125        Normalizuje sumę prawdopodobieństw do 1.0 i przycina wartości,
126        aby mieściły się w zakresie [min_prob, 1 - min_prob].
127
128        Zapobiega to zbieganiu prawdopodobieństw do 0 lub 1 zbyt szybko.
129        Minimalne prawdopodobieństwo zapewnia, że każda akcja ma zawsze
130        pewną (choćby małą) szansę na wybór.
131        """
132        min_prob = 0.001
133        p0 = self.action_probabilities[Sensor.ACTION_ACTIVE_IDX]
134        p1 = self.action_probabilities[Sensor.ACTION_SLEEP_IDX]
135
136        # Krok 1: Upewnij się, że indywidualne prawdopodobieństwa są w zakresie [0, 1]
137        p0 = max(0.0, min(1.0, p0))
138        p1 = max(0.0, min(1.0, p1))
139        
140        # Krok 2: Normalizacja sumy do 1.0
141        current_total_prob = p0 + p1
142        if abs(current_total_prob - 1.0) > 1e-9: # Jeśli suma znacząco odbiega od 1
143            if current_total_prob == 0:
144                self.action_probabilities = [0.5] * self.num_actions
145            else:
146                self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = p0 / current_total_prob
147                self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = p1 / current_total_prob
148        else:
149            self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = p0
150            self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = p1
151
152        # Krok 3: Zastosuj min_prob, zachowując sumę równą 1
153        if self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] < min_prob:
154            self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = min_prob
155            self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = 1.0 - min_prob
156        elif self.action_probabilities[Sensor.ACTION_SLEEP_IDX] < min_prob: # elif jest kluczowe
157            self.action_probabilities[Sensor.ACTION_SLEEP_IDX] = min_prob
158            self.action_probabilities[Sensor.ACTION_ACTIVE_IDX] = 1.0 - min_prob
159        
160        # Ostateczne sprawdzenie i ewentualna normalizacja (głównie na wypadek, gdyby min_prob > 0.5)
161        final_sum = sum(self.action_probabilities)
162        if abs(final_sum - 1.0) > 1e-9:
163            if final_sum == 0: self.action_probabilities = [0.5] * self.num_actions
164            else: self.action_probabilities = [p / final_sum for p in self.action_probabilities]
