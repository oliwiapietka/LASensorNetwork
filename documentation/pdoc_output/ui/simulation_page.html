 38# Wątek do uruchamiania symulacji w tle
 39class SimulationThread(QThread):
 40    """
 41    Wątek poboczny do uruchamiania logiki symulacji bez blokowania głównego wątku GUI.
 42
 43    Emituje sygnały informujące o postępie (statystyki rundy), zakończeniu
 44    symulacji (wszystkie zebrane statystyki) oraz błędach.
 45    """
 46    progress_signal = Signal(dict) # Sygnał emitujący statystyki rundy
 47    finished_signal = Signal(list) # Sygnał emitowany po zakończeniu symulacji (z wszystkimi statystykami)
 48    error_signal = Signal(str)     # Sygnał emitowany w przypadku błędu
 49
 50    def __init__(self, config_file_path):
 51        """
 52        Konstruktor SimulationThread.
 53
 54        Args:
 55            config_file_path (str): Ścieżka do pliku konfiguracyjnego dla SimulationManager.
 56            parent (QObject | None): Obiekt nadrzędny (domyślnie None).
 57        """
 58        super().__init__()
 59        self.config_file_path = config_file_path
 60        self.simulation_manager = None
 61        self._is_running = True
 62
 63    def run(self):
 64        """
 65        Metoda uruchamiana po starcie wątku (simulation_thread.start()).
 66
 67        Zawiera główną pętlę symulacji. Tworzy SimulationManager, uruchamia
 68        symulację runda po rundzie, emituje sygnały postępu i zbiera statystyki.
 69        Obsługuje zatrzymanie wątku na żądanie oraz błędy.
 70        """
 71        try:
 72            self.simulation_manager = SimulationManager(self.config_file_path)
 73            all_stats_accumulator = []
 74            for round_stats in self.simulation_manager.run_simulation():
 75                if not self._is_running:
 76                    print("SimulationThread: Stop requested.")
 77                    break
 78                self.progress_signal.emit(round_stats)
 79                all_stats_accumulator.append(round_stats)
 80            self.finished_signal.emit(all_stats_accumulator)
 81        except Exception as e:
 82            self.error_signal.emit(str(e))
 83            import traceback
 84            traceback.print_exc()
 85        finally:
 86            if self.config_file_path.startswith(tempfile.gettempdir()):
 87                try:
 88                    os.remove(self.config_file_path)
 89                    print(f"Removed temporary config file: {self.config_file_path}")
 90                except OSError as e:
 91                    print(f"Error removing temporary config file {self.config_file_path}: {e}")
 92
 93    def stop(self):
 94        """
 95        Zatrzymuje działanie wątku symulacji.
 96
 97        Ustawia wewnętrzną flagę `_is_running` na False. Jeśli obiekt
 98        SimulationManager posiada flagę `stop_simulation_flag`, ustawia
 99        ją również, aby umożliwić SimulationManagerowi czyste zakończenie
100        bieżącej rundy.
101        """
102        self._is_running = False
103        if self.simulation_manager and hasattr(self.simulation_manager, 'stop_simulation_flag'): 
104            self.simulation_manager.stop_simulation_flag = True
105        print("SimulationThread: stop() called.")

251            canvas = self.canvases[key]
252            canvas.figure.clear()

255            try:

257                if plot_method:
258                    plot_method()

260                    img_path = os.path.join(temp_plot_dir, f"{key}.png")

262                    if os.path.exists(img_path):
263                        img = plt.imread(img_path)
264                        ax.imshow(img)

266                    else:
267                        print(f"PlotResultsPage: Image file not found for {key} at {img_path}")
268                        ax.text(0.5, 0.5, f"Brak pliku wykresu dla\n{config['title']}", ha='center', va='center')

270                     ax.text(0.5, 0.5, f"Metoda dla {key} nie znaleziona", ha='center', va='center')
